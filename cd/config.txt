#!/bin/bash
set -e

# Colors
GREEN="\033[1;32m"
RED="\033[1;31m"
YELLOW="\033[1;33m"
BLUE="\033[1;34m"
NC="\033[0m"

print() { echo -e "${YELLOW}⏳ $1...${NC}"; }
success() { echo -e "${GREEN}✅ $1${NC}"; }
error() { echo -e "${RED}❌ $1${NC}"; }

# Banner
clear
echo -e "${BLUE}"
echo "   ███████╗██╗  ██╗ █████╗ ██████╗  ██████╗ ██╗    ██╗"
echo "   ██╔════╝██║  ██║██╔══██╗██╔══██╗██╔═══██╗██║    ██║"
echo "   ███████╗███████║███████║██║  ██║██║   ██║██║ █╗ ██║"
echo "   ╚════██║██╔══██║██╔══██║██║  ██║██║   ██║██║███╗██║"
echo "   ███████║██║  ██║██║  ██║██████╔╝╚██████╔╝╚███╔███╔╝"
echo "   ╚══════╝╚═╝  ╚═╝╚═╝  ╚═╝╚═════╝  ╚═════╝  ╚══╝╚══╝"
echo -e "${NC}"
echo -e "${YELLOW}        Shadow Wings Config Setup${NC}"
echo ""

# Root check
if [ "$EUID" -ne 0 ]; then
    error "Run this script as root or with sudo"
    exit 1
fi

print "Starting configuration"

# Secure input (token hidden)
read -p "🔑 Enter UUID: " UUID
read -p "🔑 Enter Token ID: " TOKEN_ID
read -s -p "🔐 Enter Token (hidden): " TOKEN
echo ""
read -p "🌐 Enter Panel URL (https://panel.domain.com): " REMOTE

# Validate inputs
if [[ -z "$UUID" || -z "$TOKEN_ID" || -z "$TOKEN" || -z "$REMOTE" ]]; then
    error "All fields are required!"
    exit 1
fi

# Create directory
print "Creating config directory"
mkdir -p /etc/pterodactyl

# Backup old config
if [ -f /etc/pterodactyl/config.yml ]; then
    cp /etc/pterodactyl/config.yml /etc/pterodactyl/config.yml.bak
    success "Old config backed up"
fi

# Write config
print "Writing config file"

tee /etc/pterodactyl/config.yml > /dev/null <<CFG
debug: false
uuid: ${UUID}
token_id: ${TOKEN_ID}
token: ${TOKEN}

api:
  host: 0.0.0.0
  port: 8080
  ssl:
    enabled: true
    cert: /etc/certs/wing/fullchain.pem
    key: /etc/certs/wing/privkey.pem
  upload_limit: 100

system:
  data: /var/lib/pterodactyl/volumes
  sftp:
    bind_port: 2022

allowed_mounts: []
remote: '${REMOTE}'
CFG

success "Config saved to /etc/pterodactyl/config.yml"

# Restart Wings
print "Starting Wings service"
systemctl restart wings
success "Wings started successfully"

# Status check
echo ""
systemctl status wings --no-pager | head -n 10

echo ""
echo -e "${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
echo -e "${GREEN}🚀 Wings is now configured!${NC}"
echo -e "${GREEN}━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━${NC}"
